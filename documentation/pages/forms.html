Form Handling

Creating Forms

<?php $form = BaseForm::getInstance(); ?>

<form action="<?= BaseRouting::url('example/store') ?>" method="POST">
    <?= $form->csrf() ?>
    
    <div class="form-group">
        <label>Name</label>
        <input type="text" 
               name="name" 
               class="<?= $form->inputClass('name') ?>" 
               value="<?= $form->input('name') ?>">
        <?= $form->error('name') ?>
    </div>
    
    <div class="form-group">
        <label>Status</label>
        <?= $form->select('status', [
            'active' => 'Active',
            'inactive' => 'Inactive'
        ], 'active', ['class' => 'form-control']) ?>
    </div>
    
    <button type="submit" class="btn btn-primary">Submit</button>
</form>

Form Validation

public function store() {
    $validation = $this->validate([
        'name' => ['required', 'min:3'],
        'email' => ['required', 'email'],
        'status' => ['required']
    ]);
    
    if ($validation !== true) {
        $form = BaseForm::getInstance();
        $form->setErrors($validation);
        return $this->redirect('example/create');
    }
    
    // Process form...
}

Available Form Methods

  • input($name, $default = '') - Get input value
  • text($name, $value, $attributes) - Create text input
  • select($name, $options, $selected, $attributes) - Create select dropdown
  • csrf() - Generate CSRF token field
  • error($field) - Get error message for field
  • inputClass($field, $defaultClass) - Get input class with validation state
