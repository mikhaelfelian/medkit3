Security Features

Input Sanitization

The framework provides automatic input sanitization through the Input library.

Basic Usage


// Get sanitized POST data
$name = $this->input->post('name');
$email = $this->input->post('email');

// Get sanitized GET data
$id = $this->input->get('id');

Security Features

  • HTML Entity Encoding
  • SQL Injection Prevention
  • Whitespace Trimming
  • Array Sanitization
  • XSS Protection

Example in Controller


public function add() {
    try {
        if ($_SERVER['REQUEST_METHOD'] === 'POST') {
            $data = [
                'kode' => $this->input->post('kode'),
                'item' => $this->input->post('item'),
                'harga' => $this->input->post('harga')
            ];
            // Process sanitized data...
        }
    } catch (Exception $e) {
        // Handle errors...
    }
}

CSRF Protection

// In your form
$form = BaseForm::getInstance();

// Automatically adds CSRF token
echo $form->open('user/store', 'POST');

// Manual CSRF token
echo $security->csrfField();

// Validate token in controller
if (!$this->security->validateRequest()) {
    throw new SecurityException('Invalid CSRF token');
}

XSS Prevention

// In views, auto-escape output
<?= $this->escape($data) ?>

// Or use the security helper
echo $security->escapeHtml($data);

// For JSON data
echo $security->escapeJson($data);

SQL Injection Prevention

// Use query builder (recommended)
$result = $this->model
    ->where('status', $status)
    ->whereIn('category', $categories)
    ->get();

// Or prepared statements
$sql = "SELECT * FROM users WHERE id = :id";
$result = $this->query($sql, [':id' => $id]);

Password Security

// Hash password
$hashedPassword = password_hash($password, PASSWORD_DEFAULT, [
    'cost' => 12
]);

// Verify password
if (password_verify($password, $hashedPassword)) {
    // Password is correct
}

Session Security

// Secure session configuration
ini_set('session.cookie_httponly', 1);
ini_set('session.use_only_cookies', 1);
ini_set('session.cookie_secure', 1);

// Session handling
session_regenerate_id(true);
$_SESSION['user_id'] = $user->id;

// Session encryption
$encrypted = $security->encrypt($data);
$decrypted = $security->decrypt($encrypted);

Input Sanitization

// Sanitize single input
$clean = $security->sanitizeInput($_POST['name']);

// Sanitize array of inputs
$data = $security->sanitizeInputs($_POST);

// Custom sanitization
$clean = $security->sanitize($input, [
    'trim',
    'strip_tags',
    'htmlspecialchars'
]);

File Upload Security

// Validate file upload
$upload = new FileUpload($_FILES['document']);

$upload->setAllowedTypes(['jpg', 'png', 'pdf'])
       ->setMaxSize(2048) // 2MB
       ->setUploadPath('uploads/')
       ->setSanitizeFilename(true);

if ($upload->save()) {
    $filename = $upload->getFilename();
} else {
    $errors = $upload->getErrors();
}

Security Headers

// Automatically set in BaseSecurity
header('X-Frame-Options: SAMEORIGIN');
header('X-XSS-Protection: 1; mode=block');
header('X-Content-Type-Options: nosniff');
header('Referrer-Policy: strict-origin-when-cross-origin');
header('Content-Security-Policy: default-src \'self\'');

Authentication

// Check authentication
if (!$this->auth->isAuthenticated()) {
    return $this->redirect('login');
}

// Check permissions
if (!$this->auth->hasPermission('admin')) {
    throw new AuthException('Unauthorized access');
}

// Role-based access
if ($this->auth->hasRole(['admin', 'manager'])) {
    // Allow access
}
