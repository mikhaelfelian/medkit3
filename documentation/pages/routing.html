Routing

Basic Routing

// config/routes.php
BaseRouting::get('example', 'Example@index');
BaseRouting::post('example/store', 'Example@store');
BaseRouting::get('example/edit/{id}', 'Example@edit');
BaseRouting::post('example/update/{id}', 'Example@update');
BaseRouting::get('example/delete/{id}', 'Example@delete');

Route Parameters

// Route with parameter
BaseRouting::get('user/{id}', 'User@show');

// In controller
public function show($id) {
    // $id is automatically passed
    $user = $this->model->find($id);
    return $this->view('user/show', ['user' => $user]);
}

Default Routes

// Set default controller and method
BaseRouting::setDefaults('Home', 'index');

// 404 handler
BaseRouting::notFound(function() {
    require_once ROOT_PATH . '/app/views/errors/404.php';
});

Route Groups

// Admin routes
if (isAdmin()) {
    BaseRouting::get('admin/dashboard', 'Admin@dashboard');
    BaseRouting::get('admin/users', 'Admin@users');
}

// API routes
BaseRouting::get('api/users', 'Api@users');
BaseRouting::post('api/users', 'Api@store');
